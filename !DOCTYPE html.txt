<canvas id="gameCanvas" width="400" height="600"></canvas>

<script>
let canvas = document.getElementById("gameCanvas");
let ctx = canvas.getContext("2d");

let player = { x:200, y:500, vy:0, color:"white" };
let gravity = 0.5;
let jumpPower = -10;

let platforms = [];
let distance = 0;
let money = 0;
let gameRunning = false;

function startGame(){
  document.getElementById("menu").style.display="none";
  canvas.style.display="block";
  resetGame();
  gameRunning = true;
  loop();
}

function resetGame(){
  player.y = 500;
  player.vy = 0;
  platforms = [];
  distance = 0;

  for(let i=0;i<6;i++){
    platforms.push({x:Math.random()*300,y:500-i*100,width:100});
  }
}

function loop(){
  if(!gameRunning) return;
  update();
  draw();
  requestAnimationFrame(loop);
}

function update(){
  player.vy += gravity;
  player.y += player.vy;

  // collision
  platforms.forEach(p=>{
    if(player.y < p.y && player.y + player.vy >= p.y &&
       player.x > p.x && player.x < p.x + p.width){
      player.vy = jumpPower;
    }
  });

  // move platforms down
  if(player.y < 300){
    player.y = 300;
    platforms.forEach(p=> p.y += 5);
    distance += 1;
  }

  // remove old platforms
  platforms = platforms.filter(p => p.y < 600);

  // add new platforms (harder)
  if(platforms.length < 6){
    let width = Math.max(40, 100 - distance/50);
    platforms.push({
      x: Math.random() * (400 - width),
      y: 0,
      width: width
    });
  }

  // reward every 100m
  if(distance % 100 === 0 && distance !== 0){
    money += 1000;
  }

  // fall
  if(player.y > 600){
    money = Math.max(0, money - 500);
    endGame();
  }
}

function draw(){
  // background change
  let level = Math.floor(distance / 100);
  let colors = ["#111","#003","#030","#300","#220","#044"];
  canvas.style.background = colors[level % colors.length];

  ctx.clearRect(0,0,400,600);

  // player
  ctx.fillStyle = player.color;
  ctx.fillRect(player.x-5, player.y-20, 10, 20);

  // platforms
  ctx.fillStyle = "white";
  platforms.forEach(p=>{
    ctx.fillRect(p.x, p.y, p.width, 10);
  });

  // UI
  ctx.fillStyle="white";
  ctx.fillText("Distance: "+distance+"m", 10,20);
  ctx.fillText("Money: $"+money, 10,40);
}

document.addEventListener("keydown", e=>{
  if(e.key === "ArrowLeft") player.x -= 20;
  if(e.key === "ArrowRight") player.x += 20;
});

function endGame(){
  gameRunning = false;
  canvas.style.display="none";
  document.getElementById("menu").style.display="block";
  document.getElementById("money").innerText = money;
}

function openShop(){
  document.getElementById("menu").style.display="none";
  document.getElementById("shop").style.display="block";
}

function backMenu(){
  document.getElementById("shop").style.display="none";
  document.getElementById("menu").style.display="block";
}

function buyColor(color, price){
  if(money >= price){
    money -= price;
    player.color = color;
    alert("Purchased!");
  } else {
    alert("Not enough money!");
  }
}
